scripted_gui = {

	prolog = {
		context_type = player_context
		
		window_name = "prolog_container"
		
		visible = {
            always = yes
		}
		
		effects = {
			prolog_open_button_click = {
				if = {
					limit = { NOT = { has_country_flag = prolog_opened } }
					set_country_flag = prolog_opened
				}
				else = {
					clr_country_flag = prolog_opened
				}
			}
		}
		
		triggers = {
			prolog_open_button_visible = {
				has_political_power < 149
			}
		}
		
		ai_test_interval = 300
		ai_test_variance = 0.1
		ai_max_weight_taken_per_test = 2
		
		ai_enabled = {
            always = yes
		}
		ai_check = {
			has_political_power > 149
		}
		
		ai_weights = {
			prolog_open_button_click = {
				weight = 1.0
				ai_will_do = {
					factor = 1
				}
			}
		}
	}
}
