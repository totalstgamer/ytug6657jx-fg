country_event = {
	id = moneymod.5 #final solution when reached maximum debt
	title = moneymod.5.t
	desc = moneymod.5.d
	picture = GFX_report_event_iww_demonstration
	is_triggered_only = yes
	
	option = {
		name = moneymod.5.a	#economycrashes
		ai_chance = { factor = 100 }	
		add_stability = -1
		add_political_power = -300
		set_variable = { var = debttotal value = 0}
		clr_country_flag = money5
	}
	option = {
		name = moneymod.5.b	#hold election
		ai_chance = { factor = 0 }	
		if={
			limit={
				OR={
					has_government = fascism
					has_government = communism
					has_government = neutrality
				}
			}
			set_variable = { var = debttotal value = 0}
			set_politics = {
    				parties = {
						democratic = { 
							popularity = 100
						}
		
						fascism = {
							popularity = 0
						}
					
						communism = {
							popularity = 0
						}
					
						neutrality = { 
							popularity = 0
						}
					}
					
				ruling_party = democratic
			}
			else={
			set_variable = { var = debttotal value = 0}
				start_civil_war = {
			    	ideology = communism
			    	size = 0.8
				}
			}
		}
		clr_country_flag = money5
	}
}
